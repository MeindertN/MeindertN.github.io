116 private void AddUnitTest(string fileName, int lineNumber, Dictionary<string, string> parameterValues, string functionName)
117 {
118 var unitTest = new UnitTestItem();
119 unitTest.UnitTestFunctionName = functionName;
120 bool identified = false;
121 string shortFileName = Path.GetFileName(fileName);
122 unitTest.UnitTestFileName = shortFileName;
123
124 foreach (var info in information)
125 {
126 if (parameterValues.ContainsKey(info.Key))
127 {
128 var value = parameterValues[info.Key];
129 //all strings are assumed to start and end with a string delimiter for all supported languages,
130 //note that for some languages the string delimiter can be """
131 if (value.StartsWith("\"\"\""))
132 value = value.Substring(3, value.Length - 6).Replace("\\\"", "\"");
133 else
134 value = value.Substring(1, value.Length - 2).Replace("\\\"", "\"");
135 switch (info.Key)
136 {
137 case "Purpose": unitTest.UnitTestPurpose = value; break;
138 case "PostCondition": unitTest.UnitTestAcceptanceCriteria = value; break;
139 case "Identifier": unitTest.ItemID = value; identified = true; break;
140 case "TraceID": unitTest.AddLinkedItem(new ItemLink(value, ItemLinkType.UnitTests)); break;
141 default: throw new Exception($"Unknown annotation identifier: {info.Key}");
142 }
143 }
144 }
145 if (!identified)
146 {
147 unitTest.ItemID = $"{shortFileName}:{lineNumber}";
148 }
149 if (gitRepo != null && !gitRepo.GetFileLocallyUpdated(fileName))
150 {
151 //if gitInfo is not null, this means some item data elements should be collected through git
152 unitTest.ItemLastUpdated = gitRepo.GetFileLastUpdated(fileName);
153 unitTest.ItemRevision = gitRepo.GetFileVersion(fileName);
154 }
155 else
156 {
157 //the last time the local file was updated is our best guess
158 unitTest.ItemLastUpdated = File.GetLastWriteTime(fileName);
159 unitTest.ItemRevision = File.GetLastWriteTime(fileName).ToString("yyyy/MM/dd HH:mm:ss");
160 }
161 if (unitTests.FindIndex(x => x.ItemID == unitTest.ItemID) != -1)
162 {
163 throw new Exception($"Duplicate unit test identifier detected in {shortFileName} in the annotation starting on line {lineNumber}. Check other unit tests to ensure all unit tests have a unique identifier.");
164 }
165 unitTests.Add(unitTest);
166 }
