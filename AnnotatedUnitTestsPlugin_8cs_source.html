75 private int ParameterEnd(string input)
76 {
77 int openers = 0;
78 int closers = 0;
79 bool insideStringLiteral = false;
80 bool ignoreStringDelim = false;
81
82 for (int i = 0; i < input.Length; i++)
83 {
84 var temp = input.Substring(i);
85 if (temp.StartsWith("\"") && !ignoreStringDelim)
86 {
87 insideStringLiteral = !insideStringLiteral;
88 }
89 ignoreStringDelim = temp.StartsWith("\\\"");
90 if (!insideStringLiteral && temp.StartsWith(parameterStartDelimiter))
91 {
92 openers++;
93 }
94 if (!insideStringLiteral && temp.StartsWith(parameterEndDelimiter))
95 {
96 closers++;
97 }
98 if (openers > 0 && openers == closers)
99 {
100 return i;
101 }
102 }
103 return -1;
104 }

106 private Dictionary<string, string> ParseParameterString(string pms, int startLine, string filename)
107 {
108 Dictionary<string, string> foundParameters = new Dictionary<string, string>();
109 //replace all = , inside the strings with non-printing replacement characters unlikely
110 //to be used in practice
111 StringBuilder pmsSb = new StringBuilder(pms);
112 bool insideString = false;
113 for (int i = 0; i < pms.Length; i++)
114 {
115 if (pms[i] == '"')
116 {
117 insideString = !insideString;
118 }
119 if (pms[i] == '=' && insideString)
120 {
121 pmsSb[i] = '\a';
122 }
123 if (pms[i] == ',' && insideString)
124 {
125 pmsSb[i] = '\f';
126 }
127 }
128 string[] parameters = pmsSb.ToString().Split(parameterSeparator);
129 foreach (var parameter in parameters)
130 {
131 string[] values = parameter.Split('=', StringSplitOptions.TrimEntries);
132 if (values.Length != 2)
133 {
134 throw new Exception($"Error parsing annotation starting on line {startLine} of \"{filename}\".");
135 }
136 var info = information.First(x => x.Value.KeyWord.ToUpper() == values[0].ToUpper());
137 if (info.Key != string.Empty)
138 {
139 StringBuilder sb = new StringBuilder(values[1]);
140 sb.Replace('\a', '=');
141 sb.Replace('\f', ',');
142 foundParameters[info.Key] = sb.ToString();
143 }
144 }
145 return foundParameters;
146 }

148 private void FindAndProcessAnnotations(string[] lines, string filename)
149 {
150 StringBuilder foundAnnotation = new StringBuilder();
151 for (int i = 0; i < lines.Length; i++)
152 {
153 int index = lines[i].IndexOf(decorationMarker, StringComparison.OrdinalIgnoreCase);
154 int paramStartIndex = -1;
155 int paramEndIndex = -1;
156 int startLine = -1;
157 if (index >= 0)
158 {
159 startLine = i;
160 foundAnnotation.Append(lines[i].Substring(index));
161 //keep iterating until we find the beginning of the parameters of the annotation
162 for (int j = i; j < lines.Length; j++)
163 {
164 paramStartIndex = foundAnnotation.ToString().IndexOf(parameterStartDelimiter);
165 if (paramStartIndex < 0)
166 {
167 foundAnnotation.Append(lines[j]);
168 }
169 else
170 {
171 i = j + 1;
172 break;
173 }
174 }
175 for (int j = i; j < lines.Length; j++)
176 {
177 paramEndIndex = ParameterEnd(foundAnnotation.ToString());
178 if (paramEndIndex >= 0)
179 {
180 break;
181 }
182 else
183 {
184 foundAnnotation.Append(lines[j]);
185 }
186 }
187 string parameterString = foundAnnotation.ToString().Substring(paramStartIndex + 1, paramEndIndex - paramStartIndex - 1);
188 foundAnnotation.Clear();
189 Dictionary<string, string> foundParameters = ParseParameterString(parameterString, startLine, filename);
190 //check if any required parameters are missing
191 foreach (var info in information)
192 {
193 if (!info.Value.Optional && !foundParameters.ContainsKey(info.Key))
194 {
195 throw new Exception($"Required parameter {info.Key} missing from unit test anotation starting on {startLine} of \"{filename}\".");
196 }
197 }
198 AddUnitTest(filename, startLine, foundParameters);
199 }
200 }

203 private void AddUnitTest(string fileName, int lineNumber, Dictionary<string, string> parameterValues)
204 {
205 var unitTest = new UnitTestItem();
206 bool identified = false;
207 string shortFileName = Path.GetFileName(fileName);
208
209 foreach (var info in information)
210 {
211 if (parameterValues.ContainsKey(info.Key))
212 {
213 var value = parameterValues[info.Key];
214 //all strings are assumed to start and end with a string delimiter for all supported languages
215 value = value.Substring(1, value.Length - 2).Replace("\\\"","\"");
216 switch (info.Key)
217 {
218 case "Purpose": unitTest.UnitTestPurpose = value; break;
219 case "PostCondition": unitTest.UnitTestAcceptanceCriteria = value; break;
220 case "Identifier": unitTest.ItemID = value; identified = true; break;
221 case "TraceID": unitTest.AddLinkedItem(new ItemLink(value, ItemLinkType.Related)); break;
222 default: throw new Exception($"Unknown annotation identifier: {info.Key}");
223 }
224 }
225 }
226 if (!identified)
227 {
228 unitTest.ItemID = $"{shortFileName}:{lineNumber}";
229 }
230 if (gitRepo != null && !gitRepo.GetFileLocallyUpdated(fileName))
231 {
232 //if gitInfo is not null, this means some item data elements should be collected through git
233 unitTest.ItemLastUpdated = gitRepo.GetFileLastUpdated(fileName);
234 unitTest.ItemRevision = gitRepo.GetFileVersion(fileName);
235 }
236 else
237 {
238 //the last time the local file was updated is our best guess
239 unitTest.ItemLastUpdated = File.GetLastWriteTime(fileName);
240 unitTest.ItemRevision = File.GetLastWriteTime(fileName).ToString("yyyy/MM/dd HH:mm:ss");
241 }
242 if (unitTests.FindIndex(x => x.ItemID == unitTest.ItemID) != -1)
243 {
244 throw new Exception($"Duplicate unit test identifier detected in {shortFileName} in the annotation starting on line {lineNumber}. Check other unit tests to ensure all unit tests have a unique identifier.");
245 }
246 unitTests.Add(unitTest);
247 }
