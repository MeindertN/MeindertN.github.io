16 public string ConvertTextile2AsciiDoc(string textile)
17 {
18 if (textile == null)
19 throw new ArgumentNullException(nameof(textile));
20
21 // --- Convert Headings ---
22 // Example: "h1. Heading" => "== Heading"
23 textile = Regex.Replace(textile, @"^h(\d)\.\s+(.*)$", m =>
24 {
25 int level = int.Parse(m.Groups[1].Value);
26 // In AsciiDoc, a level-1 section is typically "==", level-2 "===", etc.
27 string prefix = new string('=', level + 1);
28 return $"{prefix} {m.Groups[2].Value}";
29 }, RegexOptions.Multiline);
30
31 // --- Convert Links ---
32 // Textile: "link text":http://example.com
33 // AsciiDoc: link:http://example.com[link text]
34 textile = Regex.Replace(textile, @"""([^""]+)""\s*:\s*(\S+)", "link:$2[$1]");
35
36 // --- Convert Images ---
37 // Textile: !http://example.com/image.png!
38 // AsciiDoc: image::http://example.com/image.png[]
39 textile = Regex.Replace(textile, @"!(\S+)!",
40 m => $"image::{m.Groups[1].Value}[]");
41
42 // --- Convert Unordered Lists ---
43 // Instead of using spaces for nesting, output multiple '*' characters.
44 // Example: Textile "* Item" or "** Nested item" become "* Item" and "** Item"
45 textile = Regex.Replace(textile, @"^(?<stars>\*+)\s+", m =>
46 {
47 int level = m.Groups["stars"].Value.Length;
48 return new string('*', level) + " ";
49 }, RegexOptions.Multiline);
50
51 // --- Convert Ordered Lists ---
52 // Instead of using spaces for nesting, output multiple '.' characters.
53 // Example: Textile "# Item" or "## Nested item" become ". Item" and ".. Item"
54 textile = Regex.Replace(textile, @"^(?<hashes>#+)\s+", m =>
55 {
56 int level = m.Groups["hashes"].Value.Length;
57 return new string('.', level) + " ";
58 }, RegexOptions.Multiline);
59
60 // --- Convert Blockquotes (bq.) ---
61 // Textile blockquotes starting with "bq. " become AsciiDoc blockquotes.
62 textile = Regex.Replace(textile, @"^bq\.\s+(.*)$", m =>
63 "____\n" + m.Groups[1].Value + "\n____", RegexOptions.Multiline);
64
65 // --- Convert Blockquotes (lines beginning with '>') ---
66 // Lines starting with ">" are also treated as quotes.
67 textile = Regex.Replace(textile, @"^>\s*(.*)$", m =>
68 "____\n" + m.Groups[1].Value + "\n____", RegexOptions.Multiline);
69
70 // --- Convert Code Blocks ---
71 // Textile code blocks beginning with "bc. " are wrapped in AsciiDoc source block delimiters.
72 textile = Regex.Replace(textile, @"^bc\.\s+(.*)$", m =>
73 "[source]\n----\n" + m.Groups[1].Value + "\n----", RegexOptions.Multiline);
74
75 // --- Convert Inline Code ---
76 // Textile inline code marked with @ characters is converted to AsciiDoc inline code.
77 // Example: @print("hello")@ becomes `print("hello")`
78 textile = Regex.Replace(textile, @"@([^@]+)@", m =>
79 "`" + m.Groups[1].Value + "`");
80
81 // --- Convert Strikethrough ---
82 // Textile uses hyphen-delimited text for strikethrough, e.g.: -deleted text-
83 // We convert it to AsciiDocâ€™s inline strike format: [strike]#text#
84 textile = Regex.Replace(textile, @"(^|\s)-(.+?)-(?=$|\s)", m =>
85 m.Groups[1].Value + "[strike]#" + m.Groups[2].Value + "#");
86
87 // --- Process Tables ---
88 // Convert contiguous table lines into an AsciiDoc table block.
89 textile = ProcessTables(textile);
90
91 // --- Ensure List Blocks Are Preceded by a Blank Line ---
92 textile = EnsureListBlocksHaveLeadingBlankLine(textile);
93
94 return textile;
95 }

102 private string ProcessTables(string input)
103 {
104 var lines = input.Split(new[] { "\r\n", "\r", "\n" }, StringSplitOptions.None);
105 var output = new StringBuilder();
106 var tableBlock = new List<string>();
107
108 foreach (var line in lines)
109 {
110 // A simple pattern: a table row starts and ends with a pipe.
111 if (Regex.IsMatch(line, @"^\|.*\|$"))
112 {
113 tableBlock.Add(line);
114 }
115 else
116 {
117 if (tableBlock.Count > 0)
118 {
119 output.Append(ProcessTableBlock(tableBlock));
120 tableBlock.Clear();
121 }
122 output.AppendLine(line);
123 }
124 }
125 if (tableBlock.Count > 0)
126 {
127 output.Append(ProcessTableBlock(tableBlock));
128 }
129 return output.ToString();
130 }

137 private string ProcessTableBlock(List<string> tableLines)
138 {
139 var sb = new StringBuilder();
140 int numColumns = 0;
141 if (tableLines.Count > 0)
142 {
143 // Determine the number of columns from the first row.
144 var firstLine = tableLines[0].Trim();
145 // Splitting by '|' leaves empty strings at the beginning and end.
146 var cells = Regex.Split(firstLine, @"\|")
147 .Where(x => !string.IsNullOrEmpty(x))
148 .ToArray();
149 numColumns = cells.Length;
150 sb.AppendLine($"[cols=\"{numColumns}*\"]");
151 }
152 sb.AppendLine("|===");
153 foreach (var line in tableLines)
154 {
155 string trimmedLine = line.Trim();
156 // Remove the starting and ending pipe (if present)
157 if (trimmedLine.StartsWith("|"))
158 trimmedLine = trimmedLine.Substring(1);
159 if (trimmedLine.EndsWith("|"))
160 trimmedLine = trimmedLine.Substring(0, trimmedLine.Length - 1);
161 // Split the row into cells.
162 var cells = trimmedLine.Split(new char[] { '|' }, StringSplitOptions.None);
163 // Process each cell: if the cell starts with "_.", treat it as a header cell.
164 for (int i = 0; i < cells.Length; i++)
165 {
166 string cell = cells[i].Trim();
167 if (cell.StartsWith("_."))
168 {
169 // Remove the header marker and prefix with '^' for AsciiDoc header cell.
170 cell = cell.Substring(2).Trim();
171 cells[i] = $"^{cell}";
172 }
173 else
174 {
175 cells[i] = cell;
176 }
177 }
178 // In AsciiDoc, each row starts with a pipe, and cells are separated by " |"
179 sb.Append("|" + string.Join(" |", cells) + "\n");
180 }
181 sb.AppendLine("|===");
182 return sb.ToString();
183 }

191 private string EnsureListBlocksHaveLeadingBlankLine(string text)
192 {
193 // Split by newline so we can process line by line.
194 var lines = text.Split(new[] { "\r\n", "\r", "\n" }, StringSplitOptions.None);
195 var result = new List<string>();
196
197 // We'll insert a blank line before the first line of any contiguous list block,
198 // provided it isn't already preceded by a blank line.
199 for (int i = 0; i < lines.Length; i++)
200 {
201 // Detect if the current line is a list item.
202 bool isListItem = Regex.IsMatch(lines[i], @"^(?:\*+|\.+)\s+");
203 if (isListItem)
204 {
205 // If this is the first line of the file, or the previous line is not blank,
206 // and we are at the start of a list block, then insert a blank line.
207 if (i > 0 && !string.IsNullOrWhiteSpace(lines[i - 1]))
208 {
209 // Also, avoid inserting duplicate blank lines.
210 if (result.Count > 0 && !string.IsNullOrWhiteSpace(result.Last()))
211 {
212 result.Add(string.Empty);
213 }
214 }
215 }
216 result.Add(lines[i]);
217 }
218 return string.Join("\n", result);
219 }
