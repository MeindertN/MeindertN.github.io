32 private void ProcessCommands(TomlTableArray commands)
33 {
34 if (commands == null)
35 return;
36 DateTime currentDateTime = DateTime.Now;
37
38 foreach (var command in commands)
39 {
40 if (!command.ContainsKey("executable") || !command.ContainsKey("arguments") ||
41 !command.ContainsKey("workingDirectory") || !command.ContainsKey("ignoreErrors"))
42 {
43 throw new ArgumentException($"The command processor cannot process the requested command due to a missing configuration element.");
44 }
45 if ((string)command["executable"] == "") //if executable is unknown we cannot execute
46 continue;
47 executables.Add((string)command["executable"]);
48 string temp = (string)command["workingDirectory"];
49 if (temp == String.Empty)
50 {
51 temp = outputDir;
52 }
53 workingDirectories.Add(ReplaceVariables(temp, currentDateTime));
54 temp = (string)command["arguments"];
55 arguments.Add(ReplaceVariables(temp, currentDateTime));
56 ignoreErrors.Add((string)command["ignoreErrors"] == "True");
57 }
58 }

60 private string ReplaceVariables(string temp, DateTime now)
61 {
62 temp = temp.Replace("%OUTPUTFILE%", filename);
63 temp = temp.Replace("%OUTPUTDIR%", outputDir);
64 temp = temp.Replace("%OUTPUTFILENOEXT%", filenameNoExt);
65 temp = temp.Replace("%INPUTDIR%", inputDir);
66 temp = temp.Replace("%DATE%", now.ToString("yyyyMMdd"));
67 temp = temp.Replace("%DATETIME%", now.ToString("yyyyMMddHHmm"));
68 return temp;
69 }

71 public void RunCommands()

73 for (int i = 0; i < executables.Count; i++)
