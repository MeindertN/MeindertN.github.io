213 private void ScanDirectoryForConfiguration(TestConfiguration testConfig)
214 {
215 if (string.IsNullOrEmpty(testConfig.TestDirectory))
216 {
217 logger.Warn($"Empty TestDirectory in configuration for project '{testConfig.Project}', skipping");
218 return;
219 }
220
221 IDirectoryInfo dir = fileSystem.DirectoryInfo.New(testConfig.TestDirectory);

226 IFileInfo[] files = dir.GetFiles(fileMask, testConfig.SubDirs ? SearchOption.AllDirectories : SearchOption.TopDirectoryOnly);

229 testConfig.AddSourceFile(file.FullName);
230 sourceFiles.Add(file.FullName); // Keep for backward compatibility
231 logger.Debug($"Found source file: {file.FullName} (Project: {testConfig.Project}, Language: {testConfig.Language})");
232 }
233 }
234

236 }

262 {
