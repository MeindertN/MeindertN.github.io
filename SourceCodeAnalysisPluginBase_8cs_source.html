147 private static IntPtr ResolveTreeSitterNativeLibrary(string libraryName, Assembly assembly, DllImportSearchPath? searchPath)
148 {
149 try
150 {
151 var rid = GetRuntimeIdentifier();
152 var ext = GetNativeLibraryExtension();
153 var names = new List<string> { libraryName + ext };
154
155 // Non-Windows: also try lib-prefixed name (e.g., libtree-sitter-csharp.so)
156 if (!RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
157 names.Add("lib" + libraryName + ext);
158
159 // Hyphenation fallback for C#
160 if (libraryName.Equals("tree-sitter-csharp", StringComparison.OrdinalIgnoreCase))
161 {
162 names.Add("tree-sitter-c-sharp" + ext);
163 if (!RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
164 names.Add("libtree-sitter-c-sharp" + ext);
165 }
166
167 foreach (var file in names)
168 {
169 var p1 = Path.Combine(AppContext.BaseDirectory, "runtimes", rid, "native", file);
170 if (File.Exists(p1)) return NativeLibrary.Load(p1, assembly, searchPath);
171
172 var asmDir = Path.GetDirectoryName(assembly.Location);
173 if (!string.IsNullOrEmpty(asmDir))
174 {
175 var p2 = Path.Combine(asmDir, "runtimes", rid, "native", file);
176 if (File.Exists(p2)) return NativeLibrary.Load(p2, assembly, searchPath);
177 }
178 }
179 }
180 catch (Exception ex)
181 {
182 logger.Warn(ex, $"Error resolving native library '{libraryName}', falling back to default resolution");
183 }
184 return IntPtr.Zero;
185 }

187 private static string GetRuntimeIdentifier()
188 {
189 bool x64 = RuntimeInformation.ProcessArchitecture == Architecture.X64;
190 bool x86 = RuntimeInformation.ProcessArchitecture == Architecture.X86;
191 bool arm64 = RuntimeInformation.ProcessArchitecture == Architecture.Arm64;
192
193 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
194 return arm64 ? "win-arm64" : x64 ? "win-x64" : "win-x86";
195
196 if (RuntimeInformation.IsOSPlatform(OSPlatform.Linux))
197 return arm64 ? "linux-arm64" : "linux-x64";
198
199 if (RuntimeInformation.IsOSPlatform(OSPlatform.OSX))
200 return arm64 ? "osx-arm64" : "osx-x64";
201
202 throw new PlatformNotSupportedException($"Unsupported platform: {RuntimeInformation.OSDescription}");
203 }

216 return ".so"; // Default fallback
217 }
