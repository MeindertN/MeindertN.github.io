42 private void BuildDynamicTypeMap()
43 {
44 // Get all types that implement IContentCreator in the current assembly
45 var assembly = Assembly.GetAssembly(typeof(IContentCreator));
46 var contentCreatorTypes = assembly.GetTypes()
47 .Where(t => typeof(IContentCreator).IsAssignableFrom(t) && t.IsClass && !t.IsAbstract)
48 .Where(t => !_sourceTypeMap.Values.Contains(t)); // Exclude already mapped types
49
50 foreach (var type in contentCreatorTypes)
51 {
52 _dynamicTypeMap[type.Name.ToUpper()] = type;
53 }
54 }

56 public IContentCreator CreateContentCreator(DataSource source, string contentCreatorId = null)
57 {
58 // Handle special case for Comment source
59 if (source == DataSource.Comment)
60 {
61 return new CommentContentCreator();
62 }
63
64 // Try to get the type from the source map first
65 if (_sourceTypeMap.TryGetValue(source, out Type sourceType))
66 {
67 return (IContentCreator)_serviceProvider.GetRequiredService(sourceType);
68 }
69
70 var te = _traceAnalysis.GetTraceEntityForAnyProperty(contentCreatorId);
71
72 var resolveName = (te == default(TraceEntity) ? contentCreatorId : te.ID);
73
74 // For dynamic resolution (when contentCreatorId is provided)
75 if (!string.IsNullOrEmpty(resolveName))
76 {
77 if (_dynamicTypeMap.TryGetValue(resolveName.ToUpper(), out Type dynamicType))
78 {
79 return (IContentCreator)_serviceProvider.GetRequiredService(dynamicType);
80 }
81 }
82
83 throw new InvalidOperationException($"No content creator found for source '{source}' and content creator ID '{contentCreatorId}'");
84 }
