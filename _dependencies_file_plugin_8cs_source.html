98 private void ParseGradleFile(string filename)
99 {
100 if (!fileSystem.File.Exists(filename))
101 {
102 logger.Warn($"Cannot find Gradle file \"{filename}\" no dependencies will be loaded from this file.");
103 return;
104 }
105 foreach (string line in fileSystem.File.ReadLines(filename))
106 {
107 if (line.StartsWith("+---") || line.StartsWith("\\---"))
108 {
109 string[] elements = line.Substring(5).Split(':');
110 string version = elements[2].Trim();
111 bool conflict = false;
112 if (version.Contains("(*)"))
113 {
114 version = version.Split("(*)")[0].Trim();
115 }
116 if (version.Contains("->"))
117 {
118 version = version.Split("->")[0].Trim();
119 conflict = true;
120 }
121 externalDependencies.Add(new ExternalDependency($"{elements[0]}:{elements[1]}", version, conflict));
122 }
