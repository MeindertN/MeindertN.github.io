78 var filter = new Func<Commit, bool>(c =>

80 var cId = c.Tree[mdFile]?.Target?.Sha;

82 return (cId != pId);
83 });
84
85 var commits = repo.Commits.Where(filter).OrderByDescending(c => c.Author.When);
86 //return the one that is most recent and only has a single parent
