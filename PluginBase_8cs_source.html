119 private void ScrubItemsFields<T>(IEnumerable<T> items)
120 {
121 foreach (var obj in items)
122 {
123 Type type = obj.GetType();
124 PropertyInfo[] properties = type.GetProperties();
125
126 foreach (PropertyInfo property in properties)
127 {
128 if (property.PropertyType == typeof(string) && property.CanWrite)
129 {
130 // asciidoc uses | to seperate fields in a table, if the fields
131 // themselves contain a | character it needs to be escaped.
132 string currentValue = (string)property.GetValue(obj);
133 string newValue = currentValue.Replace("|", "\\|");
134 property.SetValue(obj, newValue);
135 }
136 }
137 }
138 }

142 protected void ScrubItemContents()

156 ScrubItemsFields<ExternalDependency>(dependencies);
