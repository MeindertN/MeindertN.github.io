165 if (!file.Contains(".gitignore"))

168 string destPath = fileSystem.Path.Combine(targetDir, relPath);
169 fileSystem.Directory.CreateDirectory(fileSystem.Path.GetDirectoryName(destPath));
170 fileSystem.File.Copy(file, destPath);

197 private IContentCreator GetContentObject(string contentCreatorID)
198 {
199 Assembly thisAssembly = Assembly.GetAssembly(this.GetType());
200 Type[] contentTypes = thisAssembly
201 .GetTypes()
202 .Where(t => typeof(IContentCreator).IsAssignableFrom(t) && t.IsClass)
203 .ToArray();
204
205 foreach (Type contentType in contentTypes)
206 {
207 if (contentType.Name.ToUpper() == contentCreatorID.ToUpper())
208 {
209 return Activator.CreateInstance(contentType, dataSources, traceAnalysis, configuration) as IContentCreator;
210 }
211 }
212 return null;

215 private IAISystemPlugin LoadAIPlugin()
216 {
217 if(string.IsNullOrEmpty(configuration.AIPlugin))
218 return null;
219
220 // Try loading plugins from each directory
221 foreach (var dir in configuration.PluginDirs)
222 {
223 try
224 {
225 var plugin = pluginLoader.LoadByName<IAISystemPlugin>(
226 pluginDir: dir,
227 typeName: configuration.AIPlugin,
228 configureGlobals: sc =>
229 {
230 sc.AddSingleton(fileSystem);
231 sc.AddSingleton(configuration);
232 });
233
234 if (plugin is not null)
235 {
236 logger.Info($"Found AI plugin: {plugin.Name}");
237 plugin.InitializePlugin(configuration);
238 return plugin;
239 }
240 }
241 catch (Exception ex)
242 {
243 logger.Warn($"Error loading AI plugin from directory {dir}: {ex.Message}");
244 }
245 }
246
247 logger.Warn($"Could not find AI plugin '{configuration.AIPlugin}' in any of the plugin directories.");
248 return null;
249 }
