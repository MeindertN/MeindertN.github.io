65 private List<Document> ProcessTemplates(IEnumerable<DocumentConfig> configDocuments)
66 {
67 List<Document> docs = new List<Document>();
68 foreach (var doc in configDocuments)
69 {
70 if (doc.DocumentTemplate == string.Empty)
71 continue; //skip documents without template
72 logger.Info($"Reading document template: {doc.RoboClerkID}");
73 Document document = new Document(doc.DocumentTitle, doc.DocumentTemplate);
74 document.FromStream(fileSystem.FileStream.New(fileSystem.Path.Join(configuration.TemplateDir, doc.DocumentTemplate), FileMode.Open));
75 logger.Info($"Generating document: {doc.RoboClerkID}");
76 int nrOfLevels = 0;
77 //go over the tag list to determine what information should be collected from where
78 do
79 {
80 nrOfLevels++;
81 foreach (var tag in document.RoboClerkTags)
82 {
83 if (tag.Source == DataSource.Trace)
84 {
85 logger.Debug($"Trace tag found and added to traceability: {tag.GetParameterOrDefault("ID", "ERROR")}");
86 //grab trace tag and add to the trace analysis
87 IContentCreator contentCreator = contentCreatorFactory.CreateContentCreator(DataSource.Trace, null);
88 tag.Contents = contentCreator.GetContent(tag, doc);
89 continue;
90 }
91 if (tag.Source != DataSource.Unknown)
92 {
93 try
94 {
95 IContentCreator contentCreator = contentCreatorFactory.CreateContentCreator(tag.Source, tag.ContentCreatorID);
96 tag.Contents = contentCreator.GetContent(tag, doc);
97 }
98 catch (InvalidOperationException ex)
99 {
100 logger.Warn($"Content creator for source '{tag.Source}' not found: {ex.Message}");
101 tag.Contents = $"UNABLE TO CREATE CONTENT, ENSURE THAT THE CONTENT CREATOR CLASS '{tag.Source}:{tag.ContentCreatorID}' IS KNOWN TO ROBOCLERK.\n";
102 }
103 }
104 }
105 string documentContent = document.ToText();
106 document.FromString(documentContent);
107 }
108 while (document.RoboClerkTags.Count() > 0 && nrOfLevels < 5);
109 docs.Add(document);
110 logger.Info($"Finished creating document {doc.RoboClerkID}");
111 }
112 return docs;
113 }

115 private void CleanAndCopyMediaDirectory()
116 {
117 logger.Info("Cleaning the media directory and copying the media files.");
118 string toplineDir = fileSystem.Path.GetFileName(configuration.MediaDir);
119 string targetDir = fileSystem.Path.Combine(configuration.OutputDir, toplineDir);
120 if (fileSystem.Directory.Exists(targetDir))
121 {
122 fileSystem.Directory.Delete(targetDir, true);
123 }
124 fileSystem.Directory.CreateDirectory(targetDir);
125 string[] files = fileSystem.Directory.GetFiles(configuration.MediaDir, "*", SearchOption.AllDirectories);
126 foreach (string file in files)
127 {
128 if (!file.Contains(".gitignore"))
129 {
130 var relPath = fileSystem.Path.GetRelativePath(configuration.MediaDir, file);
131 string destPath = fileSystem.Path.Combine(targetDir, relPath);
132 fileSystem.Directory.CreateDirectory(fileSystem.Path.GetDirectoryName(destPath));
133 fileSystem.File.Copy(file, destPath);
134 }
135 }
