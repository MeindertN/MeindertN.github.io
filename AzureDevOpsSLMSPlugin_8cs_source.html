175 private void AddLinksToWorkItems(IList<WorkItemRelation> links, LinkedItem item)
176 {
177 logger.Debug($"Adding links to workitem {item.ItemID}");
178 if (links != null) //check for links
179 {
180 foreach (var rel in links)
181 {
182 if (rel.Rel.Contains("Hierarchy-Forward") || rel.Rel.Contains("TestedBy-Forward"))
183 {
184 //this is a child link
185 var id = AzureDevOpsUtilities.GetWorkItemIDFromURL(rel.Url);
186 logger.Debug($"Child link found: {id}");
187 item.AddLinkedItem(new ItemLink(id, ItemLinkType.Child));
188 continue;
189 }
190 if (rel.Rel.Contains("Hierarchy-Reverse") || rel.Rel.Contains("TestedBy-Reverse"))
191 {
192 //this is a parent link
193 var id = AzureDevOpsUtilities.GetWorkItemIDFromURL(rel.Url);
194 logger.Debug($"Parent link found: {id}");
195 item.AddLinkedItem(new ItemLink(id, ItemLinkType.Parent));
196 continue;
197 }
198 if (rel.Rel.Contains(".Related"))
199 {
200 //this is a related link
201 var id = AzureDevOpsUtilities.GetWorkItemIDFromURL(rel.Url);
202 logger.Debug($"Parent link found: {id}");
203 item.AddLinkedItem(new ItemLink(id, ItemLinkType.Related));
204 continue;
205 }
206 if (rel.Rel.Contains("SharedStepReferencedBy-Reverse"))
207 {
208 //we ignore these kinds of links as this type of traceability is not of interest to RoboClerk
209 continue;
210 }
211 logger.Warn($"Unknown link type encountered in workitem {item.ItemID}: {rel.Rel}");
212 }
213 }

216 private dynamic GetWorkItemField<T>(WorkItem workitem, string field)
217 {
218 if (workitem.Fields.ContainsKey(field))
219 {
220 var ident = workitem.Fields[field] as Microsoft.VisualStudio.Services.WebApi.IdentityRef;
221 if (ident != null)
222 {
223 return ident.DisplayName;
224 }
225 else
226 {
227 return (T)workitem.Fields[field];
228 }
229 }
230 else
231 {
232 //this can also happen if a field does exist on a workitem but it has no value (i.e. the user didn't enter anything)
233 logger.Warn($"Failed to retrieve field \"{field}\" from workitem {workitem.Id} of type \"{GetWorkItemField<string>(workitem, "System.WorkItemType")}\".");
234 if (typeof(T) == typeof(string)) //ensure we return empty string instead of null
235 return string.Empty;

383 private List<TestStep> GetTestSteps(string xml, int offset = 0)
384 {
385 var result = new List<TestStep>();
386 XmlReaderSettings settings = new XmlReaderSettings();
387 settings.IgnoreWhitespace = true;
388
389 using (var memStream = new MemoryStream(Encoding.Unicode.GetBytes(xml)))
390 using (XmlReader reader = XmlReader.Create(memStream, settings))
391 {
392 var stepData = new string[2];
393 int index = 0;
394 while (reader.Read())
395 {
396 switch (reader.NodeType)
397 {
398 case XmlNodeType.Element:
399 if (reader.Name == "step")
400 {
401 index = 0;
402 }
403 if (reader.HasAttributes)
404 {
405 if (reader.Name == "compref")
406 {
407 WorkItem workitem = witClient.GetWorkItemAsync(int.Parse(reader.GetAttribute("ref")), expand: WorkItemExpand.All).Result;
408 var steps = GetTestSteps(GetWorkItemField<string>(workitem, "Microsoft.VSTS.TCM.Steps"), result.Count);
409 result.AddRange(steps);
410 }
411 }
412 break;
413 case XmlNodeType.Text:
414 if (index < 2)
415 {
416 var noHTML = Regex.Replace(reader.Value, "<[a-zA-Z/].*?>", String.Empty);
417 stepData[index] = noHTML;
418 index++;
419 if (index == stepData.Length)
420 {
421 result.Add(new TestStep((result.Count + offset + 1).ToString(), stepData[0], stepData[1]));
422 stepData = new string[2];
423 }
424 }
425 break;
426 }
427 }
428 }
429 return result;
430 }
