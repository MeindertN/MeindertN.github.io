160 protected string EscapeNonTablePipes(string text)
161 {
162 string tableBlockPattern = @"(?ms)(^\|===\s*$.*?^\|===\s*$)";
163
164 // Use Regex.Split with a capturing group so that the table blocks are kept in the result.
165 string[] segments = Regex.Split(text, tableBlockPattern);
166 var sb = new StringBuilder();
167
168 foreach (string segment in segments)
169 {
170 // If the segment itself matches our table block pattern, leave it unmodified.
171 if (Regex.IsMatch(segment, @"(?ms)^\|===\s*$.*?^\|===\s*$"))
172 {
173 sb.Append(segment);
174 }
175 else
176 {
177 // Otherwise, escape any unescaped "|" in this segment.
178 string processed = Regex.Replace(segment, @"(?<!\\‍)\|", @"\|");
179 sb.Append(processed);
180 }
181 }
182 return sb.ToString();
183 }

185 private void ScrubItemsFields<T>(IEnumerable<T> items)
186 {
187 foreach (var obj in items)
188 {
189 Type type = obj.GetType();
190 PropertyInfo[] properties = type.GetProperties();
191
192 foreach (PropertyInfo property in properties)
193 {
194 if (property.PropertyType == typeof(string) && property.CanWrite)
195 {
196 string currentValue = (string)property.GetValue(obj);
197 if (currentValue != null)
198 {
199 // Escape pipes in non-table parts of the content.
200 string newValue = EscapeNonTablePipes(currentValue);
201 property.SetValue(obj, newValue);
202 }
203 }
204 }
205 }
206 }
