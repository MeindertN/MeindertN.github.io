94 private void ScrubItemsFields<T>(IEnumerable<T> items)
95 {
96 foreach (var obj in items)
97 {
98 Type type = obj.GetType();
99 PropertyInfo[] properties = type.GetProperties();
100
101 foreach (PropertyInfo property in properties)
102 {
103 if (property.PropertyType == typeof(string) && property.CanWrite)
104 {
105 // asciidoc uses | to seperate fields in a table, if the fields
106 // themselves contain a | character it needs to be escaped.
107 string currentValue = (string)property.GetValue(obj);
108 string newValue = Regex.Replace(currentValue, "(?<!\\\\)\\|", "\\|");
109 property.SetValue(obj, newValue);
110 }
111 }
112 }
113 }
114
115 // with the exception of docContent items, all other items are visualized
116 // in tables and they need to escape the | character
117 protected void ScrubItemContents()
118 {
119 ScrubItemsFields<RequirementItem>(systemRequirements);
120 ScrubItemsFields<RequirementItem>(softwareRequirements);
121 ScrubItemsFields<RequirementItem>(documentationRequirements);
122 ScrubItemsFields<SoftwareSystemTestItem>(testCases);
123 foreach (var testCase in testCases)
124 {
125 ScrubItemsFields<TestStep>(testCase.TestCaseSteps);
126 }
127 ScrubItemsFields<AnomalyItem>(anomalies);
128 ScrubItemsFields<RiskItem>(risks);
129 ScrubItemsFields<SOUPItem>(soup);
130 ScrubItemsFields<UnitTestItem>(unitTests);
131 ScrubItemsFields<ExternalDependency>(dependencies);
