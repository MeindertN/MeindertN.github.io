100 {
101 // Get all tags in the document

123 private bool ProcessSingleTag(IRoboClerkTag tag, DocumentConfig doc)
124 {
125 if (tag.Source == DataSource.Trace)
126 {
127 logger.Debug($"Trace tag found and added to traceability: {tag.GetParameterOrDefault("ID", "ERROR")}");
128 IContentCreator contentCreator = contentCreatorFactory.CreateContentCreator(DataSource.Trace, null);
129 string newContent = contentCreator.GetContent(tag, doc);
130 if (tag.Contents != newContent)
131 {
132 tag.Contents = newContent;
133 return true;
134 }
135 return false;
136 }
137
138 if (tag.Source != DataSource.Unknown)
139 {
140 try
141 {
142 IContentCreator contentCreator = contentCreatorFactory.CreateContentCreator(tag.Source, tag.ContentCreatorID);
143 string newContent = contentCreator.GetContent(tag, doc);
144 if (tag.Contents != newContent)
145 {
146 tag.Contents = newContent;
147 return true;
148 }
149 return false;
150 }
151 catch (InvalidOperationException ex)
152 {
153 logger.Warn($"Content creator for source '{tag.Source}' not found: {ex.Message}");
154 string errorContent = $"UNABLE TO CREATE CONTENT, ENSURE THAT THE CONTENT CREATOR CLASS '{tag.Source}:{tag.ContentCreatorID}' IS KNOWN TO ROBOCLERK.\n";
155 if (tag.Contents != errorContent)
156 {
157 tag.Contents = errorContent;
158 return true;
159 }
160 return false;
161 }
162 }
163 return false;

166 private void ProcessNestedTagsRecursively(IRoboClerkTag tag, DocumentType docType)
167 {
168 if (string.IsNullOrEmpty(tag.Contents))
169 return;
170
171 // Process nested tags recursively up to 5 levels to prevent infinite loops
172 int nestedLevel = 0;
173 string currentContent = tag.Contents;
174
175 do
176 {
177 nestedLevel++;
178 var nestedTags = new List<IRoboClerkTag>();
179 bool contentChanged = false;
180
181 if (docType == DocumentType.Text)
182 {
183 // For text documents, extract text-based nested tags
184 var textBasedTags = RoboClerkTextParser.ExtractRoboClerkTags(currentContent);
185 nestedTags.AddRange(textBasedTags.Cast<IRoboClerkTag>());
186 }
187 else if (docType == DocumentType.Docx && tag is RoboClerkDocxTag docxTag)
188 {
189 // For DOCX documents, handle both embedded documents and text content
190 nestedTags.AddRange(ProcessDocxNestedContent(docxTag));
191 }
192
193 // Process each nested tag found at this level using the existing ProcessSingleTag logic
194 foreach (var nestedTag in nestedTags)
195 {
196 // Create a temporary DocumentConfig for nested processing
197 // This ensures nested tags can be processed independently
198 var tempDoc = new DocumentConfig("temp", "temp", "temp", "temp", "temp");
199
200 if (ProcessSingleTag(nestedTag, tempDoc))
201 {
202 contentChanged = true;
203 }
204 }
205
206 // If content changed, update the parent tag's content and prepare for next iteration
207 if (contentChanged)
208 {
209 if (docType == DocumentType.Text)
210 {
211 // For text documents, rebuild the content with processed nested tags
212 currentContent = ReconstructTextContentWithProcessedTags(currentContent, nestedTags.Cast<RoboClerkTextTag>().ToList());
213 tag.Contents = currentContent;
214 }
215 // For DOCX documents, the content is already updated through the tag.Contents setter
216 }
217
218 // Exit if no changes or max nested levels reached
219 if (!contentChanged || nestedLevel >= 5)
220 break;
221
222 } while (true);
223 }
