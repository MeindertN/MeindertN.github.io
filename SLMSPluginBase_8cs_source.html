164 protected void TrimLinkedItems<T>(List<T> items, List<string> retrievedIDs)
165 {
166 foreach (var item in items)
167 {
168 LinkedItem linkedItem = item as LinkedItem;
169 List<ItemLink> linkedItemsToRemove = new List<ItemLink>();
170 foreach (var itemLink in linkedItem.LinkedItems)
171 {
172 // unit tests are a special case, we cannot use this function to remove those links because
173 // the unit tests do not come from the SLMS. Otherwise this function would remove all software
174 // system tests that are "kicked" to the unit level test plan.
175 if (!retrievedIDs.Contains(itemLink.TargetID) && itemLink.LinkType != ItemLinkType.UnitTest)
176 {
177 logger.Warn($"Removing a {itemLink.LinkType} link from item \"{linkedItem.ItemID}\" to item with ID \"{itemLink.TargetID}\" because that item has a status that causes it to be ignored.");
178 linkedItemsToRemove.Add(itemLink);
179 }
180 }
181 foreach (var itemLink in linkedItemsToRemove)
182 {
183 linkedItem.RemoveLinkedItem(itemLink); //remove the link to an ignored item
184 }
185 }
186 }
