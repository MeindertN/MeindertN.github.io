97 private void ParseGradleFile(string filename)
98 {
99 if (!fileProvider.FileExists(filename))
100 {
101 logger.Warn($"Cannot find Gradle file \"{filename}\" no dependencies will be loaded from this file.");
102 return;
103 }
104 foreach (string line in fileProvider.ReadLines(filename))
105 {
106 if (line.StartsWith("+---") || line.StartsWith("\\---"))
107 {
108 string[] elements = line.Substring(5).Split(':');
109 string version = elements[2].Trim();
110 bool conflict = false;

113 version = version.Split("(*)")[0].Trim();

115 if (version.Contains("->"))
116 {
117 version = version.Split("->")[0].Trim();
118 conflict = true;
119 }
120 dependencies.Add(new ExternalDependency($"{elements[0]}:{elements[1]}", version, conflict));
121 }
