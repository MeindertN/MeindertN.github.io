22 private void LoadPlugins()
23 {
24 foreach (var val in configuration.DataSourcePlugins)
25 {
26 bool found = false;
27
28 var plugin = pluginLoader.LoadByName<IDataSourcePlugin>(
29 pluginDir: configuration.PluginDir,
30 typeName: val,
31 configureGlobals: sc =>
32 {
33 sc.AddSingleton(fileSystem);
34 sc.AddSingleton(configuration);
35 });
36 if (plugin != null)
37 {
38 plugin.InitializePlugin(configuration);
39 if (plugin as IDataSourcePlugin != null)
40 {
41 found = true;
42 var temp = plugin as IDataSourcePlugin;
43 temp.RefreshItems();
44 plugins.Add(temp);
45 }
46 }
47
48 if(!found)
49 {
50 //if we don't find a specified datasource plugin, that is reason to quit.
51 //otherwise data could be missing from the project and the user would not know.
52 throw new Exception($"Unable to find plugin {val}, please ensure the name of the plugin and plugin directories in the RoboClerk config file are correct.");
53 }
54 }
55 }
