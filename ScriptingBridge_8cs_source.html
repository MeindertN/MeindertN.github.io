194 public string EmbedAsciidocTables(string input)
195 {
196 // Split the input into lines (preserving newlines)
197 var lines = input.Split(new[] { "\r\n", "\n" }, StringSplitOptions.None);
198 var outputLines = new List<string>();
199 bool inTable = false;
200
201 foreach (var line in lines)
202 {
203 string trimmed = line.Trim();
204
205 // Detect the table delimiter
206 if (trimmed == "|===")
207 {
208 inTable = !inTable;
209 // Replace outer table delimiter with nested table delimiter
210 outputLines.Add(line.Replace("|===", "!==="));
211 }
212 else if (inTable)
213 {
214 // Preserve leading whitespace
215 int leadingSpaces = line.Length - line.TrimStart().Length;
216 string converted = line;
217
218 // If the cell begins with a pipe, replace it with an exclamation mark,
219 // but only if the pipe is not escaped.
220 if (line.TrimStart().StartsWith("|"))
221 {
222 converted = new string(' ', leadingSpaces) + "!" + line.TrimStart().Substring(1);
223 }
224
225 // Replace any unescaped cell separator pipe.
226 // The regex (?<!\\‍)\| matches any pipe that is not preceded by a backslash.
227 converted = Regex.Replace(converted, @"(?<!\\‍)\|", "!");
228 outputLines.Add(converted);
229 }
230 else
231 {
232 // Outside a table block, leave the line unchanged.
233 outputLines.Add(line);
234 }
235 }
236
237 // Rejoin all lines into a single string.
238 return string.Join("\n", outputLines);
239 }

247 public string ConvertHeadingsForTableCell(string input)
248 {
249 if (string.IsNullOrEmpty(input))
250 return input;
251
252 // Process each line to detect and transform headings
253 string[] lines = input.Split(new[] { "\r\n", "\r", "\n" }, StringSplitOptions.None);
254 for (int i = 0; i < lines.Length; i++)
255 {
256 string line = lines[i];
257
258 // Match heading patterns (e.g., "== Heading", "=== Subheading", etc.)
259 var match = System.Text.RegularExpressions.Regex.Match(line, @"^(=+)\s+(.+)$");
260 if (match.Success)
261 {
262 int level = match.Groups[1].Length;
263 string headingText = match.Groups[2].Value.Trim();
264
265 // Convert heading based on its level
266 switch (level)
267 {
268 case 1: // Document title level
269 case 2: // Top section level - convert to bold
270 lines[i] = $"*{headingText}*";
271 break;
272 case 3: // Subsection level - convert to italic
273 lines[i] = $"_{headingText}_";
274 break;
275 case 4: // Subsubsection level - convert to italic with indentation
276 lines[i] = $"&#160;&#160; _{headingText}_";
277 break;
278 case 5: // Even deeper levels - use monospace with indentation
279 case 6:
280 lines[i] = $"&#160;&#160;&#160;&#160; `{headingText}`";
281 break;
282 default: // For extremely deep levels or unexpected cases
283 lines[i] = headingText;
284 break;
285 }
286
287 // Add a blank line after the heading for better readability
288 // only if there isn't already one and we're not at the end of the text
289 if (i < lines.Length - 1 && !string.IsNullOrWhiteSpace(lines[i + 1]))
290 {
291 lines[i] += "\n";
292 }
293 }
294 }
295
296 return string.Join("\n", lines);
297 }
