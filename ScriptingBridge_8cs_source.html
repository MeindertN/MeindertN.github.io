216 public string EmbedAsciidocTables(string input)
217 {
218 // Split the input into lines (preserving newlines)
219 var lines = input.Split(new[] { "\r\n", "\n" }, StringSplitOptions.None);
220 var outputLines = new List<string>();
221 bool inTable = false;
222
223 foreach (var line in lines)
224 {
225 string trimmed = line.Trim();
226
227 // Detect the table delimiter
228 if (trimmed == "|===")
229 {
230 inTable = !inTable;
231 // Replace outer table delimiter with nested table delimiter
232 outputLines.Add(line.Replace("|===", "!==="));
233 }
234 else if (inTable)
235 {
236 // Preserve leading whitespace
237 int leadingSpaces = line.Length - line.TrimStart().Length;
238 string converted = line;
239
240 // If the cell begins with a pipe, replace it with an exclamation mark,
241 // but only if the pipe is not escaped.
242 if (line.TrimStart().StartsWith("|"))
243 {
244 converted = new string(' ', leadingSpaces) + "!" + line.TrimStart().Substring(1);
245 }
246
247 // Replace any unescaped cell separator pipe.
248 // The regex (?<!\\‍)\| matches any pipe that is not preceded by a backslash.
249 converted = Regex.Replace(converted, @"(?<!\\‍)\|", "!");
250 outputLines.Add(converted);
251 }
252 else
253 {
254 // Outside a table block, leave the line unchanged.
255 outputLines.Add(line);
256 }
257 }
258
259 // Rejoin all lines into a single string.
260 return string.Join("\n", outputLines);
261 }

269 public string ConvertHeadingsForASCIIDOCTableCell(string input)
270 {
271 if (string.IsNullOrEmpty(input))
272 return input;
273
274 // Process each line to detect and transform headings
275 string[] lines = input.Split(new[] { "\r\n", "\r", "\n" }, StringSplitOptions.None);
276 for (int i = 0; i < lines.Length; i++)
277 {
278 string line = lines[i];
279
280 // Match heading patterns (e.g., "== Heading", "=== Subheading", etc.)
281 var match = System.Text.RegularExpressions.Regex.Match(line, @"^(=+)\s+(.+)$");
282 if (match.Success)
283 {
284 int level = match.Groups[1].Length;
285 string headingText = match.Groups[2].Value.Trim();
286
287 // Convert heading based on its level
288 switch (level)
289 {
290 case 1: // Document title level
291 case 2: // Top section level - convert to bold
292 lines[i] = $"*{headingText}*";
293 break;
294 case 3: // Subsection level - convert to italic
295 lines[i] = $"_{headingText}_";
296 break;
297 case 4: // Subsubsection level - convert to italic with indentation
298 lines[i] = $"&#160;&#160; _{headingText}_";
299 break;
300 case 5: // Even deeper levels - use monospace with indentation
301 case 6:
302 lines[i] = $"&#160;&#160;&#160;&#160; `{headingText}`";
303 break;
304 default: // For extremely deep levels or unexpected cases
305 lines[i] = headingText;
306 break;
307 }
308
309 // Add a blank line after the heading for better readability
310 // only if there isn't already one and we're not at the end of the text
311 if (i < lines.Length - 1 && !string.IsNullOrWhiteSpace(lines[i + 1]))
312 {
313 lines[i] += "\n";
314 }
315 }
316 }
317
318 return string.Join("\n", lines);
319 }
