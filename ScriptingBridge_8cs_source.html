101 public string GetLinkedField(LinkedItem li, ItemLinkType linkType, bool includeTitle = true)
102 {
103 StringBuilder field = new StringBuilder();
104 var linkedItems = li.LinkedItems.Where(x => x.LinkType == linkType);
105 if (linkedItems.Count() > 0)
106 {
107 foreach (var item in linkedItems)
108 {
109 if (field.Length > 0)
110 {
111 field.Append(" / ");
112 }
113 var linkedItem = data.GetItem(item.TargetID);
114 if (linkedItem != null)
115 {
116 AddTrace(linkedItem.ItemID);
117 field.Append(linkedItem.HasLink ? $"{linkedItem.Link}[{linkedItem.ItemID}]" : linkedItem.ItemID);
118 if (includeTitle && linkedItem.ItemTitle != string.Empty)
119 {
120 field.Append($": \"{linkedItem.ItemTitle}\"");
121 }
122 }
123 else
124 {
125 logger.Error($"Item with ID {li.ItemID} has a trace link to item with ID {item.TargetID} but that item does not exist.");
126 throw new System.Exception("Item with invalid trace link encountered.");
127 }
128 }
129 return field.ToString();
130 }
131 return "N/A";
