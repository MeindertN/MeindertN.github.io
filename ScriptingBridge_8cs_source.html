249 public string EmbedAsciidocTables(string input)
250 {
251 // Split the input into lines (preserving newlines)
252 var lines = input.Split(new[] { "\r\n", "\n" }, StringSplitOptions.None);
253 var outputLines = new List<string>();
254 bool inTable = false;
255
256 foreach (var line in lines)
257 {
258 string trimmed = line.Trim();
259
260 // Detect the table delimiter
261 if (trimmed == "|===")
262 {
263 inTable = !inTable;
264 // Replace outer table delimiter with nested table delimiter
265 outputLines.Add(line.Replace("|===", "!==="));
266 }
267 else if (inTable)
268 {
269 // Preserve leading whitespace
270 int leadingSpaces = line.Length - line.TrimStart().Length;
271 string converted = line;
272
273 // If the cell begins with a pipe, replace it with an exclamation mark,
274 // but only if the pipe is not escaped.
275 if (line.TrimStart().StartsWith("|"))
276 {
277 converted = new string(' ', leadingSpaces) + "!" + line.TrimStart().Substring(1);
278 }
279
280 // Replace any unescaped cell separator pipe.
281 // The regex (?<!\\‍)\| matches any pipe that is not preceded by a backslash.
282 converted = Regex.Replace(converted, @"(?<!\\‍)\|", "!");
283 outputLines.Add(converted);
284 }
285 else
286 {
287 // Outside a table block, leave the line unchanged.
288 outputLines.Add(line);
289 }
290 }
291
292 // Rejoin all lines into a single string.
293 return string.Join("\n", outputLines);
294 }

302 public string ConvertHeadingsForTableCell(string input)
303 {
304 if (string.IsNullOrEmpty(input))
305 return input;
306
307 // Process each line to detect and transform headings
308 string[] lines = input.Split(new[] { "\r\n", "\r", "\n" }, StringSplitOptions.None);
309 for (int i = 0; i < lines.Length; i++)
310 {
311 string line = lines[i];
312
313 // Match heading patterns (e.g., "== Heading", "=== Subheading", etc.)
314 var match = System.Text.RegularExpressions.Regex.Match(line, @"^(=+)\s+(.+)$");
315 if (match.Success)
316 {
317 int level = match.Groups[1].Length;
318 string headingText = match.Groups[2].Value.Trim();
319
320 // Convert heading based on its level
321 switch (level)
322 {
323 case 1: // Document title level
324 case 2: // Top section level - convert to bold
325 lines[i] = $"*{headingText}*";
326 break;
327 case 3: // Subsection level - convert to italic
328 lines[i] = $"_{headingText}_";
329 break;
330 case 4: // Subsubsection level - convert to italic with indentation
331 lines[i] = $"&#160;&#160; _{headingText}_";
332 break;
333 case 5: // Even deeper levels - use monospace with indentation
334 case 6:
335 lines[i] = $"&#160;&#160;&#160;&#160; `{headingText}`";
336 break;
337 default: // For extremely deep levels or unexpected cases
338 lines[i] = headingText;
339 break;
340 }
341
342 // Add a blank line after the heading for better readability
343 // only if there isn't already one and we're not at the end of the text
344 if (i < lines.Length - 1 && !string.IsNullOrWhiteSpace(lines[i + 1]))
345 {
346 lines[i] += "\n";
347 }
348 }
349 }
350
351 return string.Join("\n", lines);
352 }
