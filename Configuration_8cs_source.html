148 private void ReadDocuments(TomlTable toml)
149 {
150 foreach (var doctable in (TomlTable)toml["Document"])
151 {
152 string id = (string)doctable.Key;
153
154 TomlTable doc = (TomlTable)doctable.Value;
155 if (!doc.ContainsKey("title") || !doc.ContainsKey("abbreviation") || !doc.ContainsKey("identifier"))
156 {
157 throw new Exception($"Error reading document configuration out of project configuration file for document {doctable.Key}");
158 }
159 DocumentConfig docConfig = new DocumentConfig(id, (string)doc["identifier"], (string)doc["title"], (string)doc["abbreviation"],
160 doc.ContainsKey("template") ? (string)doc["template"] : string.Empty);
161
162 if (doc.ContainsKey("template") && doc.ContainsKey("Command"))
163 {
164 docConfig.AddCommands(new Commands((TomlTableArray)doc["Command"], outputDir, Path.GetFileName((string)doc["template"]), templateDir));
165 }
166
167 documents.Add(docConfig);
168 }
169 }

171 private void ReadTruthTraceItems(TomlTable toml)
172 {
173 var truth = (TomlTable)toml["Truth"];
174 foreach (var entityTable in truth)
175 {
176 TomlTable elements = (TomlTable)entityTable.Value;
177 if (!elements.ContainsKey("name") || !elements.ContainsKey("abbreviation"))
178 {
179 throw new Exception($"Error while reading {entityTable.Key} truth entity from project config file. Check if all required elements (\"name\" and \"abbreviation\") are present.");
180 }
181 TraceEntity entity = new TraceEntity(entityTable.Key, (string)elements["name"], (string)elements["abbreviation"], TraceEntityType.Truth);
182
183 truthEntities.Add(entity);
184 }
