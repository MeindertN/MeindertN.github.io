258 private void ReadAISystemConfigurationValues(TomlTable toml)
259 {
260 if (!toml.ContainsKey("AIFeedback"))
261 {
262 logger.Warn("AIFeedback table missing from the project configuration file. RoboClerk will not be able to provide any AI feedback.");
263 return;
264 }
265 foreach (var val in (TomlTable)toml["AIFeedback"])
266 {
267 switch (val.Key)
268 {
269 case "ProvideTemplateContentFeedback":
270 checkTemplateContentsAI = ((string)val.Value).ToUpper() == "TRUE";
271 break;
272 case "TruthItemsAIFeedback":
273 {
274 foreach (var truthitemString in (TomlArray)val.Value)
275 {
276 foreach(var truthitem in truthEntities)
277 {
278 if(truthitem.ID.ToUpper() == truthitemString.ToString().ToUpper())
279 {
280 checkTraceEntitiesAI.Add(truthitem);
281 }
282 }
283 }
284 break;
285 }
286 default:
287 throw new Exception($"Unknown AIFeedback item \"{val.Key}\" found, please check project configuration file.");
288 }
289 }

314 private void ReadDocuments(TomlTable toml)
315 {
316 foreach (var doctable in (TomlTable)toml["Document"])
317 {
318 string id = (string)doctable.Key;
319
320 TomlTable doc = (TomlTable)doctable.Value;
321 if (!doc.ContainsKey("title") || !doc.ContainsKey("abbreviation") || !doc.ContainsKey("identifier"))
322 {
323 throw new Exception($"Error reading document configuration out of project configuration file for document {doctable.Key}");
324 }
325 DocumentConfig docConfig = new DocumentConfig(id, (string)doc["identifier"], (string)doc["title"], (string)doc["abbreviation"],
326 doc.ContainsKey("template") ? (string)doc["template"] : string.Empty);
327
328 if (doc.ContainsKey("template") && doc.ContainsKey("Command"))
329 {
330 docConfig.AddCommands(new Commands((TomlTableArray)doc["Command"], outputDir, Path.GetFileName((string)doc["template"]), templateDir));
331 }
332
333 documents.Add(docConfig);
334 }

337 private void ReadTruthTraceItems(TomlTable toml)
338 {
339 var truth = (TomlTable)toml["Truth"];
340 foreach (var entityTable in truth)
341 {
342 TomlTable elements = (TomlTable)entityTable.Value;
343 if (!elements.ContainsKey("name") || !elements.ContainsKey("abbreviation"))
344 {
345 throw new Exception($"Error while reading {entityTable.Key} truth entity from project config file. Check if all required elements (\"name\" and \"abbreviation\") are present.");
346 }
347 TraceEntity entity = new TraceEntity(entityTable.Key, (string)elements["name"], (string)elements["abbreviation"], TraceEntityType.Truth);
348
349 truthEntities.Add(entity);
