146 private void ReadAISystemConfigurationValues(TomlTable toml)
147 {
148 if (!toml.ContainsKey("AIFeedback"))
149 {
150 logger.Warn("AIFeedback table missing from the project configuration file. RoboClerk will not be able to provide any AI feedback.");
151 return;
152 }
153 foreach (var val in (TomlTable)toml["AIFeedback"])
154 {
155 switch (val.Key)
156 {
157 case "ProvideTemplateContentFeedback":
158 checkTemplateContentsAI = ((string)val.Value).ToUpper() == "TRUE";
159 break;
160 case "TruthItemsAIFeedback":
161 {
162 foreach (var truthitemString in (TomlArray)val.Value)
163 {
164 foreach(var truthitem in truthEntities)
165 {
166 if(truthitem.ID.ToUpper() == truthitemString.ToString().ToUpper())
167 {
168 checkTraceEntitiesAI.Add(truthitem);
169 }
170 }
171 }
172 break;
173 }
174 default:
175 throw new Exception($"Unknown AIFeedback item \"{val.Key}\" found, please check project configuration file.");
176 }
177 }

202 private void ReadDocuments(TomlTable toml)
203 {
204 foreach (var doctable in (TomlTable)toml["Document"])
205 {
206 string id = (string)doctable.Key;
207
208 TomlTable doc = (TomlTable)doctable.Value;
209 if (!doc.ContainsKey("title") || !doc.ContainsKey("abbreviation") || !doc.ContainsKey("identifier"))
210 {
211 throw new Exception($"Error reading document configuration out of project configuration file for document {doctable.Key}");
212 }
213 DocumentConfig docConfig = new DocumentConfig(id, (string)doc["identifier"], (string)doc["title"], (string)doc["abbreviation"],
214 doc.ContainsKey("template") ? (string)doc["template"] : string.Empty);
215
216 if (doc.ContainsKey("template") && doc.ContainsKey("Command"))
217 {
218 docConfig.AddCommands(new Commands((TomlTableArray)doc["Command"], outputDir, Path.GetFileName((string)doc["template"]), templateDir));
219 }
220
221 documents.Add(docConfig);
222 }
223 }

225 private void ReadTruthTraceItems(TomlTable toml)

233 throw new Exception($"Error while reading {entityTable.Key} truth entity from project config file. Check if all required elements (\"name\" and \"abbreviation\") are present.");
234 }
235 TraceEntity entity = new TraceEntity(entityTable.Key, (string)elements["name"], (string)elements["abbreviation"], TraceEntityType.Truth);
236
237 truthEntities.Add(entity);
238 }
