72 string pluginDir,
73 string typeName,
74 Action<IServiceCollection>? configureGlobals = null
75 ) where TPluginInterface : class, IPlugin
76 {
77 // 1) Build the container and capture the list of discovered impl types:
78 var (services, implTypes) = BuildContainer<TPluginInterface>(pluginDir, configureGlobals);
79 var provider = services.BuildServiceProvider();
80
81 // 2) Find the one whose class name matches
82 var match = implTypes

84 if (match is null)

91 // -------------------------------------------------
92 // INTERNAL: assemble IServiceCollection + impl types
93 // -------------------------------------------------
94 private (IServiceCollection services, List<Type> implTypes) BuildContainer<TPluginInterface>
95 (string pluginDir,Action<IServiceCollection>? configureGlobals) where TPluginInterface : class, IPlugin
96 {
97 if (!_fileSystem.Directory.Exists(pluginDir))
98 throw new DirectoryNotFoundException($"Plugin directory not found: {pluginDir}");
99
100 var services = new ServiceCollection();
101 var implTypes = new List<Type>();
102
103 // 1) globals
104 services.AddSingleton<IFileSystem>(_fileSystem);
105 configureGlobals?.Invoke(services);
106
107 // 2) per‐assembly scan
108 foreach (var asm in _assemblyLoader.LoadFromDirectory(pluginDir))
109 {
110 var pluginTypes = asm
111 .GetTypes()
112 .Where(t => typeof(TPluginInterface).IsAssignableFrom(t)
113 && !t.IsAbstract
114 && t.GetConstructor(new[] { typeof(IFileSystem) }) != null);
115
116 foreach (var type in pluginTypes)
117 {
118 ConstructorInfo? ctor = null;
119 object?[] args;
120
121 // 3) find the single‐arg ctor
122 ctor = type.GetConstructor(new[] { typeof(IFileSystem) });
123
124 if (ctor != null)
125 {
126 args = new object[] { _fileSystem };
127 }
128 else
129 {
130 // Fallback to parameterless constructor
131 ctor = type.GetConstructor(Type.EmptyTypes);
132 if (ctor == null)
133 throw new InvalidOperationException($"Type {type.FullName} has no supported constructor.");
134
135 args = Array.Empty<object>();
136 }
137
138 implTypes.Add(type);
139
140 // 4) invoke it to get the PluginBase/IPluginRegistrar
141 var metadataInstance = (TPluginInterface)ctor.Invoke(args);
142
143 // 5) let the plugin register everything it needs,
144 metadataInstance.ConfigureServices(services);
145 }
146 }
147
148 return (services, implTypes);
149 }
