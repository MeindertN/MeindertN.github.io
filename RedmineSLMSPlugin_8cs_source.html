278 private (List<T>, List<T>) CheckForLinkedItem<T>(List<string> retrievedIDs, List<T> inputItems, List<ItemLinkType> lt) where T : LinkedItem
279 {
280 List<T> items = new List<T>();
281 List<T> removedItems = new List<T>();
282 string removeItem = string.Empty;
283 foreach (var item in inputItems)
284 {
285 if(item.LinkedItems.Count() > 0) //orphan items without links are always included so they don't get lost or ingnored.
286 removeItem = item.ItemID;
287 foreach (var link in item.LinkedItems)
288 {
289 if (link != null && lt.Contains(link.LinkType) &&
290 retrievedIDs.Contains(link.TargetID) &&
291 (systemRequirements.Any(obj => obj.ItemID == link.TargetID) ||
292 softwareRequirements.Any(obj => obj.ItemID == link.TargetID) ||
293 documentationRequirements.Any(obj => obj.ItemID == link.TargetID)) )
294 {
295 removeItem = string.Empty;
296 break;
297 }
298 }
299 if (removeItem == string.Empty)
300 {
301 items.Add(item);
302 }
303 else
304 {
305 logger.Warn($"Removing {item.ItemType} {item.ItemID} because item(s) {string.Join(',',item.LinkedItems.Select(item => item.TargetID))} it was originally linked to is(are) not valid.");
306 removedItems.Add(item);
307 retrievedIDs.Remove(removeItem);
308 removeItem = string.Empty;
309 }
310 }
311 return (items, removedItems);
312 }

362 protected SoftwareSystemTestItem CreateTestCase(List<RedmineIssue> issues, RedmineIssue redmineItem)

368 resultItem.ItemID = redmineItem.Id.ToString();

371 resultItem.ItemStatus = redmineItem.Status.Name ?? string.Empty;
372 resultItem.TestCaseState = redmineItem.Status.Name ?? string.Empty;
373 resultItem.ItemTitle = redmineItem.Subject ?? string.Empty;

377 }

379 {
380 resultItem.Link = new Uri($"{baseURL}{resultItem.ItemID}");
381 }
382 logger.Debug($"Getting test steps for item: {redmineItem.Id}");
383 string itemDescription = redmineItem.Description ?? string.Empty;
384 var testCaseSteps = SoftwareSystemTestItem.GetTestSteps(convertTextile ? converter.ConvertTextile2AsciiDoc(itemDescription) : itemDescription);
385 foreach (var testCaseStep in testCaseSteps)
386 {
387 resultItem.AddTestCaseStep(testCaseStep);
388 }
389 resultItem.TestCaseAutomated = false;
390 if (redmineItem.CustomFields != null)
391 {
392 foreach (var field in redmineItem.CustomFields)
393 {
394 if (field.Value != null)
395 {
396 var value = (System.Text.Json.JsonElement)field.Value;
397 if (field.Name == "Test Method")
398 {
399 resultItem.TestCaseAutomated = (value.GetString() == "Automated") || (value.GetString() == "Unit Tested");
400 resultItem.TestCaseToUnitTest = (value.GetString() == "Unit Tested");
401 }
402 if (field.Name == "Identifier") //check if this test case is being kicked to the unit test plan
403 {
404 string values = value.GetString();
405 var unitTests = values.Split(',');
406 foreach(var unitTest in unitTests)
407 {
408 resultItem.KickToUnitTest(unitTest.Trim());
409 }
410 }
411 }
412 }
413 }
414
415 AddLinksToItem(redmineItem, resultItem);
416 //ensure that the relationships to requirements are of type "Tests"
417 foreach (var link in resultItem.LinkedItems)
418 {
419 foreach (var issue in issues)
420 {
421 if (issue.Id.ToString() == link.TargetID && issue.Tracker.Name == SrsConfig.Name)
422 {
423 link.LinkType = ItemLinkType.Tests;
424 break;
425 }
426 }
427 }
428
429 return resultItem;
430 }

723 protected bool ShouldIgnoreIssue(RedmineIssue redmineItem, TruthItemConfig config, out string reason)
724 {
725 reason = string.Empty;
726 if (!config.Filtered)
727 return false;
728
729 var properties = redmineItem.GetType().GetProperties();
730 foreach (var property in properties)
731 {
732 if (property.Name != "CustomFields" &&
733 property.Name != "EstimatedHours" &&
734 property.Name != "Relations")
735 {
736 HashSet<string> values = new HashSet<string>();
737 var value = property.GetValue(redmineItem);
738 if (value != null)
739 {
740 if (value is string || value is int || value is bool)
741 {
742 values.Add(value.ToString());
743 }
744 else if (value is DateTime date)
745 {
746 values.Add(date.ToString("MM-dd-yyyy"));
747 }
748 else
749 {
750 PropertyInfo nameProperty = value.GetType().GetProperty("Name");
751 if (nameProperty != null && nameProperty.PropertyType == typeof(string))
752 {
753 values.Add(nameProperty.GetValue(value) as string);
754 }
755 }
756 if (!IncludeItem(property.Name, values))
757 {
758 reason = $"Item does not match inclusion filter for \"{property.Name}\" with value \"{String.Join(", ", values)}\"";
759 logger.Warn($"Ignoring {redmineItem.Tracker.Name} item {redmineItem.Id} due to \"{property.Name}\" not matching inclusion filter.");
760 return true;
761 }
762 if (ExcludeItem(property.Name, values))
763 {
764 reason = $"Item matches exclusion filter for \"{property.Name}\" with value \"{String.Join(", ", values)}\"";
765 logger.Warn($"Ignoring {redmineItem.Tracker.Name} item {redmineItem.Id} due to \"{property.Name}\" matching exclusion filter.");
766 return true;
767 }
768 }
769 }
770 }
771 if (redmineItem.CustomFields.Count != 0)
772 {
773 foreach (var field in redmineItem.CustomFields)
774 {
775 if (field.Value != null)
776 {
777 bool versionField = redmineVersionFields.Contains(field.Name);
778 HashSet<string> values = new HashSet<string>();
779 if (field.Multiple)
780 {
781 foreach (var value in (System.Text.Json.JsonElement.ArrayEnumerator)field.Value)
782 {
783 values.Add(ConvertValue(versionField, value.ToString()));
784 }
785 }
786 else
787 {
788 values.Add(ConvertValue(versionField, ((System.Text.Json.JsonElement)field.Value).GetString()));
789 }
790 if (!IncludeItem(field.Name, values))
791 {
792 reason = $"Item does not match inclusion filter for custom field \"{field.Name}\" with value \"{String.Join(", ", values)}\"";
793 logger.Warn($"Ignoring {redmineItem.Tracker.Name} item {redmineItem.Id} due to \"{field.Name}\" not matching inclusion filter.");
794 return true;
795 }
796 if (ExcludeItem(field.Name, values))
797 {
798 reason = $"Item matches exclusion filter for custom field \"{field.Name}\" with value \"{String.Join(", ", values)}\"";
799 logger.Warn($"Ignoring {redmineItem.Tracker.Name} item {redmineItem.Id} due to \"{field.Name}\" matching exclusion filter.");
800 return true;
801 }
802 }
803 }
804 }
805 return false;
806 }

808 protected RequirementItem CreateRequirement(List<RedmineIssue> issues, RedmineIssue redmineItem, RequirementType requirementType)
809 {
810 logger.Debug($"Creating requirement item: {redmineItem.Id}");
811 RequirementItem resultItem = new RequirementItem(requirementType);
812
813 resultItem.ItemProject = redmineItem.Project.Name ?? string.Empty;
814 resultItem.ItemCategory = "Unknown";
815 if (redmineItem.CustomFields.Count != 0)
816 {
817 foreach (var field in redmineItem.CustomFields)
818 {
819 if (field.Name == "Functional Area" && field.Value != null)
820 {
821 resultItem.ItemCategory = ((System.Text.Json.JsonElement)field.Value).GetString();
822 }
823 }
824 }
825
826 if (redmineItem.AssignedTo != null)
827 {
828 resultItem.RequirementAssignee = redmineItem.AssignedTo.Name;
829 }

831 {
832 resultItem.RequirementAssignee = string.Empty;

835 resultItem.RequirementDescription = convertTextile ? converter.ConvertTextile2AsciiDoc(itemDescription) : itemDescription;
836 resultItem.ItemID = redmineItem.Id.ToString();
837 resultItem.ItemRevision = redmineItem.UpdatedOn.ToString();
838 resultItem.ItemLastUpdated = (DateTime)redmineItem.UpdatedOn;
839 if (redmineItem.FixedVersion != null)
840 {
841 resultItem.ItemTargetVersion = redmineItem.FixedVersion.Name ?? string.Empty;
842 }
843 resultItem.ItemStatus = redmineItem.Status.Name ?? string.Empty;
844 resultItem.RequirementState = redmineItem.Status.Name ?? string.Empty;
845 resultItem.ItemTitle = redmineItem.Subject ?? string.Empty;
846 resultItem.TypeOfRequirement = requirementType;
847 if (baseURL != "")
848 {
849 resultItem.Link = new Uri($"{baseURL}{resultItem.ItemID}");
850 }
851
852 foreach (var issue in issues)
853 {
854 if (issue.Parent != null && issue.Parent.Id == redmineItem.Id)
855 {
856 resultItem.AddLinkedItem(new ItemLink(issue.Id.ToString(), ItemLinkType.Child));
857 }
858 }
859
860 AddLinksToItem(redmineItem, resultItem);
861
862 //ensure that the relationships to SoftwareSystemTests are of type "TestedBy"
863 foreach (var link in resultItem.LinkedItems)
864 {
865 foreach (var issue in issues)
866 {
867 if (issue.Id.ToString() == link.TargetID && issue.Tracker.Name == TcConfig.Name)
868 {
869 link.LinkType = ItemLinkType.TestedBy;
870 }
871 }
872 }
