133 private List<TestStep> GetTestSteps(string testDescription)
134 {
135 string[] lines = testDescription.Split('\n');
136 List<TestStep> output = new List<TestStep>();
137 bool thenFound = false;
138 foreach (var line in lines)
139 {
140 if (string.IsNullOrWhiteSpace(line))
141 {
142 continue; //skip empty lines
143 }
144 if (!line.ToUpper().Contains("THEN:") && !thenFound)
145 {
146 output.Add(new TestStep((output.Count + 1).ToString(), line, string.Empty));
147 }
148 else
149 {
150 if (!thenFound)
151 {
152 thenFound = true;
153 output[output.Count - 1].ExpectedResult = line;
154 }
155 else if (!line.ToUpper().Contains("AND:"))
156 {
157 output[output.Count - 1].ExpectedResult = output[output.Count - 1].ExpectedResult + '\n' + line;
158 }
159 else
160 {
161 output.Add(new TestStep((output.Count + 1).ToString(), string.Empty, line));
162 }
163 }
164 }
165 return output;
166 }

545 private int GetProjectID(string projectName)
546 {
547 var request = new RestRequest("projects.json", Method.Get)
548 .AddParameter("limit", 100)
549 .AddParameter("key", apiKey);
550 List<RedmineProject> projects = new List<RedmineProject>();
551 var response = client.GetAsync<RedmineProjects>(request).GetAwaiter().GetResult();
552 projects.AddRange(response.Projects);
553 while (response.Limit + response.Offset < response.TotalCount)
554 {
555 request.AddOrUpdateParameter("offset", response.Offset + response.Limit);
556 response = client.GetAsync<RedmineProjects>(request).GetAwaiter().GetResult();
557 projects.AddRange(response.Projects);
558 }
559 foreach (var project in projects)
560 {
561 if (project.Name == projectName)
562 {
563 logger.Info($"Found project \"{projectName}\" in Redmine. Description: {project.Description}, ID#: {project.Id}.");
564 return project.Id;
565 }
566 }
567 throw new Exception($"Could not find project \"{projectName}\" in Redmine. Please check plugin configuration file and Redmine server.");
