53 private static Dictionary<string, string> GetConfigOptions(IEnumerable<string> commandlineOptions, ILogger logger)
54 {
55 Dictionary<string, string> options = new Dictionary<string, string>();
56 foreach (var commandlineOption in commandlineOptions)
57 {
58 if (commandlineOption != ",")
59 {
60 var elements = commandlineOption.Split('=');
61 if (elements.Length != 2)
62 {
63 logger.Error($"Commandline option can not be parsed: {commandlineOption}. Please check commandline call, it should be in the form of <IDENTIFIER>=<VALUE>");
64 Console.WriteLine($"An error occurred parsing commandline option: {commandlineOption}. Expected syntax is <IDENTIFIER>=<VALUE>.");
65 throw new Exception("Error parsing commandline options.");
66 }
67 options[elements[0]] = elements[1];
68 }
69 }
70 return options;
71 }

131 private static IAISystemPlugin LoadAIPlugin(IConfiguration config, IPluginLoader pluginLoader)
132 {
133 // Try loading plugins from each directory
134 var logger = NLog.LogManager.GetCurrentClassLogger();
135 foreach (var dir in config.PluginDirs)
136 {
137 try
138 {
139 var plugin = pluginLoader.LoadByName<IAISystemPlugin>(
140 pluginDir: dir,
141 typeName: config.AIPlugin,
142 configureGlobals: sc =>
143 {
144 sc.AddSingleton(config);
145 });
146
147 if (plugin is not null)

149 logger.Info($"Found AI plugin: {plugin.Name}");
150 plugin.InitializePlugin(config);

152 }
153 }
154 catch (Exception ex)

156 logger.Warn($"Error loading AI plugin from directory {dir}: {ex.Message}");
157 }
158 }
