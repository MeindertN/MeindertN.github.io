55 internal static Dictionary<string, string> GetConfigOptions(IEnumerable<string> commandlineOptions, ILogger logger)
56 {
57 Dictionary<string, string> options = new Dictionary<string, string>();
58 foreach (var commandlineOption in commandlineOptions)
59 {
60 if (commandlineOption != ",")
61 {
62 var elements = commandlineOption.Split('=');
63 if (elements.Length != 2)
64 {
65 logger.Error($"Commandline option can not be parsed: {commandlineOption}. Please check commandline call, it should be in the form of <IDENTIFIER>=<VALUE>");
66 Console.WriteLine($"An error occurred parsing commandline option: {commandlineOption}. Expected syntax is <IDENTIFIER>=<VALUE>.");
67 throw new Exception("Error parsing commandline options.");
68 }
69 options[elements[0]] = elements[1];
70 }
71 }
72 return options;
73 }

89 internal static void RegisterContentCreators(IServiceCollection services)
90 {
91 ArgumentNullException.ThrowIfNull(services);
92
93 // Get the assembly containing the content creators
94 var currentDir = Path.GetDirectoryName(Assembly.GetExecutingAssembly().Location)!;
95 var otherAssemblyPath = Path.Combine(currentDir, "RoboClerk.Core.dll");
96 var assembly = Assembly.LoadFrom(otherAssemblyPath);
97
98 // Find all types that implement IContentCreator
99 var contentCreatorTypes = assembly.GetTypes()
100 .Where(t => typeof(IContentCreator).IsAssignableFrom(t) &&
101 !t.IsInterface &&
102 !t.IsAbstract &&
103 !t.IsGenericType)
104 .ToList();
105
106 var logger = NLog.LogManager.GetCurrentClassLogger();
107 logger.Debug($"Found {contentCreatorTypes.Count} content creator types to register");
108
109 foreach (var type in contentCreatorTypes)

111 try
112 {
113 // Register each content creator as transient
114 services.AddTransient(type);
115 logger.Debug($"Registered content creator: {type.Name}");

137 internal static IAISystemPlugin LoadAIPlugin(IConfiguration config, IPluginLoader pluginLoader)
138 {
139 // Try loading plugins from each directory
140 var logger = NLog.LogManager.GetCurrentClassLogger();
141 foreach (var dir in config.PluginDirs)
142 {
143 IAISystemPlugin plugin = null;
144 try
145 {
146 plugin = pluginLoader.LoadByName<IAISystemPlugin>(
147 pluginDir: dir,
148 typeName: config.AIPlugin,
149 configureGlobals: sc =>
150 {
151 sc.AddSingleton(config);
152 });
153 }
154 catch (Exception ex)
155 {
156 logger.Warn($"Error loading AI plugin from directory {dir}: {ex.Message}. Will try other directories.");
157 }
158 try
159 {
160 if (plugin is not null)
161 {
162 logger.Info($"Found AI plugin: {plugin.Name}");
163 plugin.InitializePlugin(config);
164 return plugin;
165 }
166 }
167 catch (Exception ex)
168 {
169 logger.Warn($"Error initializing AI plugin from directory {dir}: {ex.Message}");
170 return null;
171 }
172 }
173 logger.Warn($"Could not find AI plugin '{config.AIPlugin}' in any of the plugin directories.");
174 return null;
175 }
