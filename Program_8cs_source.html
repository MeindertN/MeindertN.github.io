25 internal static Dictionary<string, string> GetConfigOptions(IEnumerable<string> commandlineOptions)
26 {
27 Dictionary<string, string> options = new Dictionary<string, string>();
28 foreach (var commandlineOption in commandlineOptions)
29 {
30 if (commandlineOption != ",")
31 {
32 var elements = commandlineOption.Split('=');
33 if (elements.Length != 2)
34 {
35 Console.WriteLine($"An error occurred parsing commandline option: {commandlineOption}. Expected syntax is <IDENTIFIER>=<VALUE>.");
36 throw new Exception("Error parsing commandline options.");
37 }
38 options[elements[0]] = elements[1];
39 }
40 }
41 return options;
42 }

105 internal static IAISystemPlugin LoadAIPlugin(IConfiguration config, IPluginLoader pluginLoader)
106 {
107 // Try loading plugins from each directory
108 var logger = NLog.LogManager.GetCurrentClassLogger();
109 foreach (var dir in config.PluginDirs)
110 {
111 IAISystemPlugin plugin = null;
112 try
113 {
114 plugin = pluginLoader.LoadByName<IAISystemPlugin>(
115 pluginDir: dir,
116 typeName: config.AIPlugin,
117 configureGlobals: sc =>
118 {
119 sc.AddSingleton(config);
120 });
121 }
122 catch (Exception ex)
123 {
124 logger.Warn($"Error loading AI plugin from directory {dir}: {ex.Message}. Will try other directories.");
125 }
126 try
127 {
128 if (plugin is not null)
129 {
130 logger.Info($"Found AI plugin: {plugin.Name}");
131 plugin.InitializePlugin(config);
132 return plugin;
133 }
134 }
135 catch (Exception ex)
136 {
137 logger.Warn($"Error initializing AI plugin from directory {dir}: {ex.Message}");
138 return null;
139 }
140 }
141 logger.Warn($"Could not find AI plugin '{config.AIPlugin}' in any of the plugin directories.");
142 return null;
