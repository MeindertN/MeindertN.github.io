172 private void ProcessRoboClerkContainerInlineTag(int startIndex, int endIndex, string rawDocument)
173 {
174 tagStart = startIndex;
175 tagEnd = endIndex + 1;
176 contentStart = startIndex + 2; //remove starting tag
177 contentEnd = endIndex; //remove ending tag
178 contents = rawDocument.Substring(contentStart, contentEnd - contentStart);
179 try
180 {
181 ValidateTagContents(contents);
182 }
183 catch (TagInvalidException e)
184 {
185 e.SetLocation(tagStart, rawDocument);
186 throw;
187 }
188 ExtractParameters(contents);
189 var items = contents.Split('(')[0].Split(':');
190 contentCreatorID = items[1].Trim();
191 source = GetSource(items[0].Trim().ToUpper());
192 if (source == DataSource.Unknown)
193 {
194 throw new TagInvalidException(contents, $"Unknown datasource {items[0].Trim()}");
195 }

198 private void ValidateTagContents(string tagContents)
199 {
200 //verify required elements are present and in the right order
201 if (tagContents.Count(f => (f == '(')) != 1 ||
202 tagContents.Count(f => (f == ')')) != 1 ||
203 tagContents.Substring(0, tagContents.IndexOf('(')).Count(f => (f == ':')) != 1 ||
204 tagContents.IndexOf(')') < tagContents.IndexOf('(') ||
205 tagContents.IndexOf(':') > tagContents.IndexOf('('))
206 {
207 throw new TagInvalidException(tagContents, "RoboClerk tag is not formatted correctly");
208 }
209 //verify the preamble
210 string temp = Regex.Replace(tagContents, @"\s+", ""); //remove whitespace
211 string[] preamble = temp.Split('(')[0].Split(':', StringSplitOptions.RemoveEmptyEntries);
212 if (preamble.Length != 2)
213 {
214 throw new TagInvalidException(tagContents, "Preamble section in RoboClerk tag not formatted correctly");
215 }
216 //verify the parameter string
217 if (temp.IndexOf(')') - temp.IndexOf('(') > 1)
218 {
219 if (temp.Count(f => (f == '=')) - temp.Count(f => (f == ',')) != 1)
220 {
221 throw new TagInvalidException(tagContents, "Parameter section in RoboClerk tag not formatted correctly");
222 }
223 //isolate the parameter string and check each individual element
224 string contents = temp.Split('(')[1].Split(')')[0];
225 string[] elements = contents.Split(',');
226 foreach (var element in elements)
227 {
228 if (element.Count(f => (f == '=')) != 1)
229 {
230 throw new TagInvalidException(tagContents, "Malformed element in parameter section of RoboClerk tag");
231 }
232 string[] variables = element.Split('=', StringSplitOptions.RemoveEmptyEntries);
233 if (variables.Length != 2)
234 {
235 throw new TagInvalidException(tagContents, "Malformed element in parameter section of RoboClerk tag");
236 }
237 }
238 }

241 private void ProcessRoboClerkContainerTag(int startIndex, int endIndex, string rawDocument)
242 {
243 tagStart = startIndex;
244 tagEnd = endIndex + 3;
245 //parse the tagInfo, items are separated by :
246 string tagContents = rawDocument.Substring(startIndex + 3, endIndex - startIndex).Split('\n')[0];
247 try
248 {
249 ValidateTagContents(tagContents);
250 }
251 catch (TagInvalidException e)
252 {
253 e.SetLocation(tagStart, rawDocument);
254 throw;
255 }
256 var items = tagContents.Split(':');
257 source = GetSource(items[0].Trim().ToUpper());
258 if(source == DataSource.Unknown)
259 {
260 throw new TagInvalidException(tagContents, $"Unknown datasource {items[0].Trim()}");
261 }
262 contentCreatorID = items[1].Split('(')[0].Trim();
263 ExtractParameters(tagContents);
264 var prelimTagContents = rawDocument.Substring(startIndex, endIndex - startIndex + 1);
265 contentStart = startIndex + prelimTagContents.IndexOf('\n') + 1; //ensure to skip linebreak
266 if (prelimTagContents.IndexOf('\n') == prelimTagContents.LastIndexOf('\n'))
267 {
268 //this tag is empty
269 contentEnd = contentStart - 1;
270 contents = "";
271 }
272 else
273 {
274 contentEnd = startIndex + prelimTagContents.LastIndexOf('\n');
275 contents = rawDocument.Substring(contentStart, contentEnd - contentStart + 1);
276 }
277 }

279 private void ExtractParameters(string parameterString)
280 {
281 int paramStart = parameterString.IndexOf('(');
282 int paramEnd = parameterString.IndexOf(')');
283 if (paramEnd - paramStart == 1)
284 {
285 return; //there are no parameters
286 }
287 string param = parameterString.Substring(paramStart + 1, paramEnd - paramStart - 1);
288 var items = param.Split(',');
289 if (items.Length > 0)
290 {
291 foreach (var item in items)
292 {
293 parameters[item.Split('=')[0].Trim().ToUpper()] = item.Split('=')[1].Trim();
294 }
295 }
