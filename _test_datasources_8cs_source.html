376 private bool CompareObjects<T>(List<T> obj1, List<T> obj2)
377 {
378 var properties = typeof(T).GetProperties();
379 if(obj1.Count != obj2.Count)
380 {
381 return false;
382 }
383 for (int i = 0; i < obj1.Count; i++)
384 {
385 foreach (var prop in properties)
386 {
387 if (prop.GetValue(obj1[i]) as IEnumerable<dynamic> == null)
388 {
389 if (prop.GetValue(obj1[i]) != null && prop.GetValue(obj2[i]) != null) //skip items that are null
390 if (!prop.GetValue(obj1[i]).Equals(prop.GetValue(obj2[i])))
391 {
392 return false;
393 }
394 }
395 else
396 {
397 if((prop.GetValue(obj1[i]) as IEnumerable<dynamic>).Count() != (prop.GetValue(obj2[i]) as IEnumerable<dynamic>).Count())
398 {
399 return false;
400 }
401 }
402 }
403 }
404 return true;
405 }
